# Libre!
country_event = {
	id = flavor_trn.500
	title = "flavor_trn.EVTNAME500"
	desc = "flavor_trn.EVTDESC500"
	picture = COURT_eventPicture
	
	fire_only_once = yes

	trigger = {
		tag = TRN
		is_subject = no
		is_great_power = yes
		government = monarchy
		religion = catholic
		OR = {
			num_of_allies = 0
			every_ally = {
				religion = catholic
				NOT = { government = republic }
			}
		}
		NOT = { num_of_loans = 1 }
		all_owned_province = {
			limit = { region = iberia_region }
			religion = catholic
		}
		ruler_is_foreigner = no
		has_foreign_heir = no
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = "flavor_trn.EVTOPTA500"
		add_country_modifier = {
			name = free_spain
			duration = 7300
		}
	}
}
